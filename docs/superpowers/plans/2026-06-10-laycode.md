> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a mobile app (Expo RN) + bridge server (Node.js) that lets users control opencode from their phone while lying in bed.

**Architecture:** Bridge server runs on the desktop, proxies `/opencode-api/*` to opencode on `localhost:4096`, adds token auth and mDNS discovery. Expo RN app connects to bridge over LAN, provides a mobile-optimized UI for sessions, code viewing, and file browsing.

**Tech Stack:** Bridge: Node.js + TypeScript + express + multicast-dns + ws. App: Expo (managed) + React Navigation + @opencode-ai/sdk.

```ts
import { BridgeConfig } from './types.js'

export function parseArgs(): BridgeConfig {
  const args = process.argv.slice(2)
  const config: BridgeConfig = {
    token: '',
    port: 8079,
    opencodeUrl: 'http://localhost:4096',
  }

  for (let i = 0; i < args.length; i++) {
    switch (args[i]) {
      case '--token':
        config.token = args[++i]
        break
      case '--port':
        config.port = parseInt(args[++i], 10)
        break
      case '--opencode-url':
        config.opencodeUrl = args[++i]
        break
    }
  }

  if (!config.token) {
    console.error('Error: --token is required')
    process.exit(1)
  }

  return config
}

export function printStartupInfo(config: BridgeConfig) {
  console.log(`LayCode Bridge v0.1.0`)

  console.log(`  OpenCode:    ${config.opencodeUrl}`)
```

```ts
import { Request, Response, NextFunction } from 'express'

export function createAuthMiddleware(token: string) {
  return (req: Request, res: Response, next: NextFunction) => {
    const auth = req.headers.authorization

    if (!auth || !auth.startsWith('Bearer ') || auth.slice(7) !== token) {
      res.status(401).json({ error: 'Unauthorized' })
      return
    }

    next()
  }
}
```

```ts
import { Request, Response } from 'express'
import { BridgeConfig } from './types.js'

export function createProxyHandler(config: BridgeConfig) {
  return async (req: Request, res: Response) => {
    const targetPath = req.path.replace(/^\/opencode-api/, '')
    const targetUrl = `${config.opencodeUrl}${targetPath}${req.url.includes('?') ? '?' + req.url.split('?')[1] : ''}`

    // Build headers, forwarding auth token from APP
    const headers: Record<string, string> = {}
    for (const [key, value] of Object.entries(req.headers)) {
      if (['host', 'connection'].includes(key)) continue
      if (value) headers[key] = Array.isArray(value) ? value.join(', ') : value
    }

    try {
      const response = await fetch(targetUrl, {
        method: req.method,
        headers: {
          ...headers,
          'host': new URL(config.opencodeUrl).host,
        },
        body: req.method !== 'GET' && req.method !== 'HEAD' ? JSON.stringify(req.body) : undefined,
      })

      // Forward status and headers
      res.status(response.status)
      for (const [key, value] of response.headers) {
        if (key !== 'content-encoding') res.setHeader(key, value)
      }

      // Stream body
      if (response.body) {
        const reader = response.body.getReader()
        const pump = async () => {
          while (true) {
            const { done, value } = await reader.read()
            if (done) break
            res.write(value)
          }
          res.end()
        }
        pump().catch(() => res.end())
      } else {
        res.end()
      }
    } catch (err) {
      console.error('Proxy error:', err)
      res.status(502).json({ error: 'Bad Gateway', message: 'Cannot reach opencode server' })
    }
  }
}
```

```ts
import multicastDns from 'multicast-dns'

let mdns: ReturnType<typeof multicastDns> | null = null

export function startMdns(port: number) {
  mdns = multicastDns()

  mdns.on('query', (query) => {
    if (query.questions?.some(q => q.name === '_laycode._tcp.local')) {
      mdns?.respond({
        answers: [{
          name: '_laycode._tcp.local',
          type: 'SRV',
          ttl: 120,
          data: {
            port,
            target: require('os').hostname() + '.local',
          },
        }, {
          name: '_laycode._tcp.local',
          type: 'TXT',
          ttl: 120,
          data: require('os').hostname(),
        }],
      })
    }
  })

  console.log(`  mDNS:        _laycode._tcp advertising on port ${port}`)
}

export function stopMdns() {
  mdns?.destroy()
  mdns = null
}
```

```ts
import { WebSocketServer, WebSocket } from 'ws'
import { createServer } from 'http'
import { BridgeConfig } from './types.js'

let wss: WebSocketServer | null = null

export function startWebSocketServer(config: BridgeConfig) {
  const server = createServer()
  wss = new WebSocketServer({ server })

  wss.on('connection', (ws) => {
    // Connect to opencode SSE event stream
    const sseUrl = `${config.opencodeUrl}/event`
    const controller = new AbortController()

    const connectSSE = async () => {
      try {
        const response = await fetch(sseUrl, {
          headers: { 'Accept': 'text/event-stream' },
          signal: controller.signal,
        })

        const reader = response.body!.getReader()
        const decoder = new TextDecoder()
        let buffer = ''

        while (true) {
          const { done, value } = await reader.read()
          if (done) break

          buffer += decoder.decode(value, { stream: true })
          const lines = buffer.split('\n')
          buffer = lines.pop() || ''

          for (const line of lines) {
            if (line.startsWith('data: ')) {
              if (ws.readyState === WebSocket.OPEN) {
                ws.send(line.slice(6))
              }
            }
          }
        }
      } catch (err: any) {
        if (err.name !== 'AbortError') {
          console.error('SSE connection error:', err)
        }
      }
    }

    connectSSE()

    ws.on('close', () => controller.abort())
    ws.on('error', () => controller.abort())
  })

  wss.on('error', (err) => {
    console.error('WebSocket error:', err)
  })

  // Share the same HTTP server — combine with express
  // For now, use a separate port (config.port + 1)
  const wsPort = config.port + 1
  server.listen(wsPort, () => {
    console.log(`  WebSocket:   ws://0.0.0.0:${wsPort}/event`)
  })
}

export function stopWebSocketServer() {
  wss?.close()
  wss = null
}
```

```ts
import { ServerConfig, Project, Session, Message, FileEntry } from '../types'

export class LayCodeClient {
  private baseUrl: string
  private token: string

  constructor(config: ServerConfig) {
    this.baseUrl = `http://${config.host}:${config.port}`
    this.token = config.token
  }

  private headers() {
    return {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${this.token}`,
    }
  }

  private async request<T>(path: string, options?: RequestInit): Promise<T> {
    const res = await fetch(`${this.baseUrl}/opencode-api${path}`, {
      ...options,
      headers: { ...this.headers(), ...options?.headers },
    })
    if (!res.ok) throw new Error(`API error: ${res.status}`)
    return res.json()
  }

  // Projects
  async listProjects(): Promise<Project[]> {
    return this.request('/project')
  }

  // Sessions
  async listSessions(projectId: string): Promise<Session[]> {
    return this.request(`/project/${projectId}/session`)
  }

  async createSession(projectId: string, directory: string): Promise<Session> {
    return this.request(`/project/${projectId}/session`, {
      method: 'POST',
      body: JSON.stringify({ directory }),
    })
  }

  async getMessages(projectId: string, sessionId: string): Promise<Message[]> {
    return this.request(`/project/${projectId}/session/${sessionId}/message`)
  }

  async sendMessage(projectId: string, sessionId: string, text: string): Promise<Message> {
    return this.request(`/project/${projectId}/session/${sessionId}/message`, {
      method: 'POST',
      body: JSON.stringify({ parts: [{ type: 'text', text }] }),
    })
  }

  // Files
  async listFiles(projectId: string, path: string = '/'): Promise<FileEntry[]> {
    return this.request(`/project/${projectId}/session/find/file?path=${encodeURIComponent(path)}`)
  }

  async readFile(projectId: string, path: string): Promise<string> {
    const res = await this.request<{ content: string }>(
      `/project/${projectId}/session/file/content?path=${encodeURIComponent(path)}`
    )
    return res.content
  }

  // Health check
  async health(): Promise<boolean> {
    try {
      const res = await fetch(`${this.baseUrl}/api/v1/health`)
      return res.ok
    } catch {
      return false
    }
  }
}
```

```tsx
  route: any
  themeMode: ThemeMode

  const theme = getTheme(themeMode)
  const [messages, setMessages] = useState<Message[]>([])

  const [sending, setSending] = useState(false)

    const text = input.trim()
    setInput('')

    setMessages(prev => [...prev, userMsg])

    try {
      const reply = await client.sendMessage(projectId, sessionId, text)
      setMessages(prev => [...prev, reply])
    } catch (err) {
      console.error(err)
    }
    setSending(false)
  }

  return (
    <SafeAreaView style={[styles.container, { backgroundColor: theme.background }]} edges={['top']}>
      <FlatList
        ref={flatListRef}
        data={messages}
        keyExtractor={item => item.id}
        renderItem={({ item }) => <MessageBubble message={item} theme={theme} />}
        style={styles.list}
        onContentSizeChange={() => flatListRef.current?.scrollToEnd()}

            onPress={handleSend}
            disabled={sending}
          >
            <Text style={styles.sendText}>发送</Text>
          </TouchableOpacity>
        </View>
      </KeyboardAvoidingView>
    </SafeAreaView>
  )
}
```

```tsx
        <Text style={[styles.fileContent, { color: theme.text }]}>{fileContent}</Text>

      <View style={styles.header}>
        <Text style={[styles.title, { color: theme.text }]}>文件</Text>
```
